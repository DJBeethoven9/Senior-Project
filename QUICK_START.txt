#include "human_detector.h"

void loop() {
  // Call this every loop - it handles detection
  humanDetectorUpdate();
  
  // Check if person is present
  if (isHumanPresent()) {
    Serial.println("Person in room!");
    digitalWrite(LED_PIN, HIGH);  // Turn on LED
  } else {
    Serial.println("Room empty");
    digitalWrite(LED_PIN, LOW);   // Turn off LED
  }
}

// ============================================
// 9. TROUBLESHOOTING
// ============================================

Problem: System stuck on "CALIBRATING"
→ Make sure room is empty during calibration
→ Check CSI is enabled: "CSI enabled!" in serial

Problem: Always detects humans (false positives)
→ Increase MOTION_THRESHOLD (less sensitive)
→ Increase MIN_DETECTION_FRAMES

Problem: Never detects humans (false negatives)
→ Decrease MOTION_THRESHOLD (more sensitive)
→ Decrease MIN_DETECTION_FRAMES
→ Move more around the room during test

Problem: Weak WiFi signal
→ Move ESP32 closer to router
→ Check antenna connection
→ Look for interference (2.4 GHz conflict)
